### Javascript
```javascript
function initEventsCalendar() {
  const calendars = document.querySelectorAll("[data-events-calendar-init]");
  if (!calendars.length) return;
  const settings = {
    maxMonthsPast: 12,
    maxMonthsFuture: 12,
    translations: {
      months: ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"],
      weekdays: {
        monday: "M",
        tuesday: "T",
        wednesday: "W",
        thursday: "T",
        friday: "F",
        saturday: "S",
        sunday: "S"
      }
    }
  };
  calendars.forEach((calendar) => {
    const timezone = calendar.getAttribute("data-events-calendar-timezone") || "Europe/Amsterdam";
    const grid = calendar.querySelector("[data-events-calendar-grid]");
    const monthLabel = calendar.querySelector("[data-events-calendar-month]");
    const prevButton = calendar.querySelector("[data-events-calendar-prev]");
    const nextButton = calendar.querySelector("[data-events-calendar-next]");
    const weekdayElements = [...calendar.querySelectorAll("[data-events-calendar-weekday]")];
    const allCards = [...calendar.querySelectorAll("[data-events-calendar-card-date]")];
    const emptyState = calendar.querySelector("[data-events-calendar-empty-status]");
    const templateTiles = [...grid.querySelectorAll("[data-events-calendar-tile]")];
    if (!templateTiles.length) return;
    const templateTile = templateTiles[0].cloneNode(true);
    const now = getNowInTimezone(timezone);
    const eventsMap = createEventsMap();
    let currentMonthDate = new Date(now.year, now.month, 1);
    let selectedDate = getInitialSelectedDate();
    if (selectedDate) {
      const selectedDateObject = parseDateKey(selectedDate);
      currentMonthDate = new Date(selectedDateObject.getFullYear(), selectedDateObject.getMonth(), 1);
    }
    renderWeekdays();
    renderCalendar();
    updateCards();
    prevButton.addEventListener("click", () => {
      const targetDate = new Date(currentMonthDate.getFullYear(), currentMonthDate.getMonth() - 1, 1);
      if (!canGoToMonth(targetDate)) return;
      currentMonthDate = targetDate;
      renderCalendar();
    });
    nextButton.addEventListener("click", () => {
      const targetDate = new Date(currentMonthDate.getFullYear(), currentMonthDate.getMonth() + 1, 1);
      if (!canGoToMonth(targetDate)) return;
      currentMonthDate = targetDate;
      renderCalendar();
    });
    function renderWeekdays() {
      weekdayElements.forEach((weekday) => {
        const day = weekday.getAttribute("data-events-calendar-weekday");
        if (settings.translations.weekdays[day]) weekday.textContent = settings.translations.weekdays[day];
      });
    }
    function renderCalendar() {
      const year = currentMonthDate.getFullYear();
      const month = currentMonthDate.getMonth();
      monthLabel.textContent = \`${settings.translations.months[month]} ${year}\`;
      grid.innerHTML = "";
      const firstDay = new Date(year, month, 1);
      const firstWeekday = getWeekdayName(firstDay.getDay());
      const weekdayOrder = getWeekdayOrder();
      const startDay = weekdayOrder.indexOf(firstWeekday);
      const daysInMonth = new Date(year, month + 1, 0).getDate();
      updateButtonStates();
      for (let i = 0; i < 42; i++) {
        const tile = templateTile.cloneNode(true);
        const dayElement = tile.querySelector("[data-events-calendar-tile-day]");
        const countElement = tile.querySelector("[data-events-calendar-tile-count]");
        const dayNumber = i - startDay + 1;
        tile.removeAttribute("data-events-calendar-date");
        tile.removeAttribute("data-events-calendar-today");
        tile.removeAttribute("data-events-calendar-events-count");
        if (countElement) {
          countElement.setAttribute("data-events-calendar-tile-count", "");
          countElement.textContent = "";
        }
        if (dayNumber < 1 || dayNumber > daysInMonth) {
          tile.setAttribute("data-events-calendar-tile-status", "empty");
          dayElement.textContent = "";
          grid.appendChild(tile);
          continue;
        }
        const currentDate = new Date(year, month, dayNumber);
        const dateKey = formatDateKey(currentDate);
        const hasEvents = eventsMap[dateKey];
        const isSelected = selectedDate === dateKey;
        const isToday = dateKey === formatDateKey(new Date(now.year, now.month, now.day));
        tile.setAttribute("data-events-calendar-date", dateKey);
        dayElement.textContent = dayNumber;
        if (isSelected) {
          tile.setAttribute("data-events-calendar-tile-status", "selected");
        } else if (hasEvents) {
          tile.setAttribute("data-events-calendar-tile-status", "event");
        } else {
          tile.setAttribute("data-events-calendar-tile-status", "no-event");
        }
        if (isToday) tile.setAttribute("data-events-calendar-today", "");
        if (hasEvents) {
          tile.setAttribute("data-events-calendar-events-count", hasEvents.length);
          if (countElement) {
            countElement.setAttribute("data-events-calendar-tile-count", hasEvents.length);
            countElement.textContent = hasEvents.length;
          }
        }
        tile.addEventListener("click", () => {
          if (!eventsMap[dateKey] || selectedDate === dateKey) return;
          selectedDate = dateKey;
          updateCards();
          updateTileStates();
        });
        grid.appendChild(tile);
      }
    }
    function updateTileStates() {
      const tiles = [...grid.querySelectorAll("[data-events-calendar-date]")];
      tiles.forEach((tile) => {
        const dateKey = tile.getAttribute("data-events-calendar-date");
        const hasEvents = eventsMap[dateKey];
        if (selectedDate === dateKey) {
          tile.setAttribute("data-events-calendar-tile-status", "selected");
        } else if (hasEvents) {
          tile.setAttribute("data-events-calendar-tile-status", "event");
        } else {
          tile.setAttribute("data-events-calendar-tile-status", "no-event");
        }
      });
    }
    function updateCards() {
      allCards.forEach((card) => card.setAttribute("data-events-calendar-card-status", "not-active"));
      if (emptyState) emptyState.setAttribute("data-events-calendar-empty-status", "not-active");
      if (!selectedDate) {
        if (emptyState) emptyState.setAttribute("data-events-calendar-empty-status", "active");
        return;
      }
      const matchingCards = allCards.filter((card) => {
        const cardDate = normalizeCardDate(card.getAttribute("data-events-calendar-card-date"));
        return cardDate === selectedDate;
      });
      if (!matchingCards.length) {
        if (emptyState) emptyState.setAttribute("data-events-calendar-empty-status", "active");
        return;
      }
      matchingCards.forEach((card) => {
        card.setAttribute("data-events-calendar-card-status", "active");
        card.setAttribute("data-events-calendar-active-count", matchingCards.length);
      });
    }
    function updateButtonStates() {
      const prevDate = new Date(currentMonthDate.getFullYear(), currentMonthDate.getMonth() - 1, 1);
      const nextDate = new Date(currentMonthDate.getFullYear(), currentMonthDate.getMonth() + 1, 1);
      setButtonState(prevButton, canGoToMonth(prevDate));
      setButtonState(nextButton, canGoToMonth(nextDate));
    }
    function setButtonState(button, isActive) {
      if (!button) return;
      button.disabled = !isActive;
      button.setAttribute("aria-disabled", isActive ? "false" : "true");
    }
    function canGoToMonth(date) {
      const minDate = new Date(now.year, now.month - settings.maxMonthsPast, 1);
      const maxDate = new Date(now.year, now.month + settings.maxMonthsFuture, 1);
      return date >= minDate && date <= maxDate;
    }
    function createEventsMap() {
      const map = {};
      allCards.forEach((card) => {
        const rawDate = card.getAttribute("data-events-calendar-card-date");
        const normalizedDate = normalizeCardDate(rawDate);
        if (!normalizedDate) return;
        if (!map[normalizedDate]) map[normalizedDate] = [];
        map[normalizedDate].push(card);
      });
      return map;
    }
    function getInitialSelectedDate() {
      const todayKey = formatDateKey(new Date(now.year, now.month, now.day));
      const eventDates = Object.keys(eventsMap).sort();
      if (eventDates.includes(todayKey)) return todayKey;
      return eventDates.find((date) => date >= todayKey) || null;
    }
    function getWeekdayOrder() {
      return weekdayElements.map((weekday) => {
        return weekday.getAttribute("data-events-calendar-weekday");
      });
    }
    function getWeekdayName(dayIndex) {
      return ["sunday", "monday", "tuesday", "wednesday", "thursday", "friday", "saturday"][dayIndex];
    }
    function normalizeCardDate(dateString) {
      const parsedDate = new Date(dateString);
      if (isNaN(parsedDate)) return null;
      return formatDateKey(parsedDate);
    }
    function parseDateKey(dateKey) {
      const [year, month, day] = dateKey.split("-").map(Number);
      return new Date(year, month - 1, day);
    }
    function formatDateKey(date) {
      const year = date.getFullYear();
      const month = String(date.getMonth() + 1).padStart(2, "0");
      const day = String(date.getDate()).padStart(2, "0");
      return \`${year}-${month}-${day}\`;
    }
    function getNowInTimezone(timezone) {
      const formatter = new Intl.DateTimeFormat("en-US", {
        timeZone: timezone,
        year: "numeric",
        month: "numeric",
        day: "numeric"
      });
      const parts = formatter.formatToParts(new Date());
      const values = {};
      parts.forEach((part) => values[part.type] = part.value);
      return {
        year: Number(values.year),
        month: Number(values.month) - 1,
        day: Number(values.day)
      };
    }
  });
}
// Initialize Events Calendar (Date Picker)
document.addEventListener("DOMContentLoaded", () => {
  initEventsCalendar();
});
```

### Implementation
#### Container
Use `[data-events-calendar-init]` on the main wrapper to initialize one calendar instance, allowing multiple calendars on the same page.
Optionally you can add `[data-events-calendar-timezone]` to define the timezone used for today, the current month, and the first upcoming event. You can find a full list with timezone (TZ Identifier) options on [Wikipedia](https://en.wikipedia.org/wiki/List_of_tz_database_time_zones)
### HTML
```text
<div data-events-calendar-init data-events-calendar-timezone="Europe/Amsterdam">
  ...
</div>
```
#### Month
Use `[data-events-calendar-month]` on the text element that displays the active month and year.
The script automatically updates the month name using the `translations.months` settings array.
### Javascript
```javascript
const settings = {
  ...
  translations: {
    months: ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"],
    ...
  }
};
```
#### Navigation
Use `[data-events-calendar-prev]` and `[data-events-calendar-next]` on the buttons that navigate to the previous or next month.
The script automatically adds the native `disabled` attribute when the minimum or maximum allowed month has been reached.
#### Month Range
Use `maxMonthsPast` and `maxMonthsFuture` inside the settings object to control how many months users can navigate backward or forward from the current month.
### Javascript
```javascript
const settings = {
  maxMonthsPast: 12,
  maxMonthsFuture: 12,
  ...
};
```
#### Weekday
Use `[data-events-calendar-weekday="monday"]` through `[data-events-calendar-weekday="sunday"]` on each weekday label, where the order in the HTML decides the first day of the week.
### HTML
```text
<div data-events-calendar-weekday="monday"></div>
<div data-events-calendar-weekday="tuesday"></div>
<div data-events-calendar-weekday="wednesday"></div>
<div data-events-calendar-weekday="thursday"></div>
<div data-events-calendar-weekday="friday"></div>
<div data-events-calendar-weekday="saturday"></div>
<div data-events-calendar-weekday="sunday"></div>
```
The script automatically updates the text content of each weekday using the `translations.weekdays` settings object.
### Javascript
```javascript
const settings = {
  ...
  translations: {
    ...
    weekdays: {
      monday: "M",
      tuesday: "T",
      wednesday: "W",
      thursday: "T",
      friday: "F",
      saturday: "S",
      sunday: "S"
    }
  }
};
```
#### Weekday Order (Sunday First)
Moving `[data-events-calendar-weekday="sunday"]` to the first position (order in the HTML) automatically changes the calendar to start on Sunday instead of Monday.
#### Grid
Use `[data-events-calendar-grid]` on the calendar grid wrapper where the script clones and renders all 42 calendar tiles.
#### Tile (Template)
Use `[data-events-calendar-tile]` on the template tile inside the grid, which the script clones to create all days for the active month.
Only a single tile is required in the initial HTML structure, allowing all additional placeholder tiles to be removed from Webflow/HTML before publishing.
### HTML
```text
<div data-events-calendar-grid>
  <button data-events-calendar-tile data-events-calendar-tile-status="event" class="events-calendar__tile">
    <span data-events-calendar-tile-day class="events-calendar__tile-span">6</span>
    <span data-events-calendar-tile-count="" class="events-calendar__tile-count">2</span>
  </button>
</div>
```
#### Tile Day
Use `[data-events-calendar-tile-day]` inside each tile to display the day number.
#### Tile Count
Use `[data-events-calendar-tile-count]` inside each tile to display the amount of events on that day.
#### Tile Status
The script adds `[data-events-calendar-tile-status="empty/no-event/event/selected"]` to each tile based on its current state.
- `empty` is used for filler tiles before or after the current month inside the fixed 42 tile grid.
- `no-event` is used for real days inside the current month that do not contain any events.
- `event` is used for days that contain one or more linked event cards and can be clicked to activate those cards.
- `selected` is used for the currently active event day connected to the visible active cards.
#### Today
The script adds `[data-events-calendar-today="true"]` to the tile that matches today in the selected timezone.
#### Event Count
The script adds `[data-events-calendar-events-count="2"]` to a tile when one or more cards match that date.
#### Card Date
Use `[data-events-calendar-card-date="May 16, 2026"]` on each event card to connect it to a specific calendar day.
The date must use a readable date format supported by the JavaScript `Date()` parser, where `"May 16, 2026"` is the recommended format for the most reliable results.
### HTML
```text
<div data-events-calendar-card-date="May 16, 2026" data-events-calendar-card-status="not-active">
  <!-- Event Card Content -->
</div>
```
#### Card Status
The script adds `[data-events-calendar-card-status="active"]` or `[data-events-calendar-card-status="not-active"]` to each card based on the selected event date.
#### Active Count
The script adds `[data-events-calendar-active-count="2"]` to active cards when multiple cards share the selected date.
#### First Upcoming Event
The calendar automatically selects the first upcoming event on load, including today when an event exists on the current day.
If the next event exists in another month, the calendar automatically opens on that month.
#### No Upcoming Events
The script activates `[data-events-calendar-empty-status="active"]` when there are no upcoming events available.
### HTML
```text
<div data-events-calendar-empty-status="not-active">
  <!-- No Upcoming Events Card Content -->
</div>
```
